                if (dizi[j] == temp) {

    public int karsilastirFutbolcu(int BkartListIndex,int KkartListIndex,int pozisyon){
    switch (pozisyon) {
        case 0:
            if(FkartListesi[KkartListIndex].getPenalti()>FkartListesi[BkartListIndex].getPenalti())
                return 1;
            else if(FkartListesi[KkartListIndex].getPenalti()==FkartListesi[BkartListIndex].getPenalti())
                return 0;
            else
                return -1;
        case 1:
            if(FkartListesi[KkartListIndex].getSetbestVurus()>FkartListesi[BkartListIndex].getSetbestVurus())
                return 1;
            else if(FkartListesi[KkartListIndex].getSetbestVurus()==FkartListesi[BkartListIndex].getSetbestVurus())
                return 0;
            else
                return -1;
        default:
            if(FkartListesi[KkartListIndex].getKaleciKarsiKarsiya()>FkartListesi[BkartListIndex].getKaleciKarsiKarsiya())
                return 1;
            else if(FkartListesi[KkartListIndex].getKaleciKarsiKarsiya()==FkartListesi[BkartListIndex].getKaleciKarsiKarsiya())
                return 0;
            else
                return -1;
    }
    }
